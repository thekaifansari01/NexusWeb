⚛️ React / Next.js

Drop-in component for your React application.

React · Next.js · Gatsby
JavaScript (React) 
// 1. Create NexusWidget.jsx
import { useEffect } from 'react';

export default function NexusWidget() {
  useEffect(() => {
    window.NexusConfig = {
      apiKey: 'YOUR_NEXUS_API_KEY',
      model: 'meta-llama/llama-4-scout-17b-16e-instruct',
      theme: 'auto'
    };
    const script = document.createElement('script');
    script.src = 'https://cdn.jsdelivr.net/npm/nexus-web-assistant@3/dist/nexus-assistant.min.js';
    script.async = true;
    script.defer = true;
    document.body.appendChild(script);
  }, []);

  return null;
}

// 2. Use it anywhere:
// <NexusWidget />

Works with Next.js App Router and Pages Router. Add to _app.js or layout.js.

🔷 WordPress

Add Nexus to your WordPress site in 2 clicks.

PHP · WordPress
PHP (functions.php) 
// Add to functions.php or use a Code Snippets plugin
function add_nexus_ai_widget() {
    if (!is_admin()) {
        ?>
        <script>
        window.NexusConfig = {
            apiKey: 'YOUR_NEXUS_API_KEY',
            theme: 'auto'
        };
        </script>
        <script defer src="https://cdn.jsdelivr.net/npm/nexus-web-assistant@3/dist/nexus-assistant.min.js"></script>
        <?php
    }
}
add_action('wp_head', 'add_nexus_ai_widget');

Alternatively, install Nexus AI Widget plugin from WordPress.org (coming soon).

🟨 Vanilla JS

Basic script tag for any HTML website.

HTML · JavaScript
HTML 
<!-- Add this to your HTML, preferably before </body> -->
<script>
window.NexusConfig = {
    apiKey: 'YOUR_NEXUS_API_KEY',
    model: 'meta-llama/llama-4-scout-17b-16e-instruct',
    botName: 'Nexus AI',
    greeting: '👋 Hello! Ask me anything about this page.',
    theme: 'auto'
};
</script>
<script defer src="https://cdn.jsdelivr.net/npm/nexus-web-assistant@3/dist/nexus-assistant.min.js"></script>

Works on any static site, Ghost, Jekyll, or plain HTML. No build tools needed.
